#include <stdio.h>

int main(){
	pid_t pid = fork();

  • You can check processes in ps -a or ps -aux
  • fork() creates something called child process
  • Calling process is called the parent process
  • Memory mappings and unmappings of one of the process does not affect the parent process
  • Child and parent same
    • Child has unique uid
    • Child’s parent PID same as parent PID
  • 0 PID means child
  • +ve number means parent
  • ./process.out &runs the process in the background and you can check ps and see the processes
  • killall -9 process.out → kills all processes running this program

pid_t wait()

  • If any of the process status changes, wait checks
  • If wait is not executed, the child process does not clear the memory unless the parent itself is dead
#include <sys/types.h>
#include <unistd.h>
#include <stdio.h>

int main(){
	pid_t pid = fork();
	int n;
	if(pid == 0){
		n = 5;
		printf("Child Process: %d, Parent Process: %d\n", getpid(), getppid());
		printf("Child: n: %d\n", n);
	}else{
		for(;;);
		// wait(NULL);
		n = 15;
		printf("Parent Process: %d, Child Process: %d\n", getpid(), pid);
		printf("Parent: n: %d\n", n);
	}
	
	return 0;
}
  • If we inspect ps now, we get defunct processes → parent not ended but child ended
  • If we want grandchild to wait for child but not the parent to wait
#include <sys/types.h>
#include <sys/wait.h>
#include <unistd.h>
#include <stdio.h>

int main(){
        pid_t pid = fork();
        int n;
        if(pid == 0){
                n = 5;
                printf("Child Process: %d, Parent Process: %d\n", getpid(), getppid());
                printf("Child: n: %d\n", n);

                pid_t gc = fork();
                if(gc == 0){
                        printf("Grand Child Process: %d, Child Process: %d\n", getpid(), getppid());
                        for(;;);
                        printf("Terminated Grand Child\n");
                }else{
                        printf("Child Process: %d, Grand Child Process: %d\n", getpid(), pid);
                        waitpid(pid, NULL, 0);
                        printf("Terminated Child\n");
                }
        }else{
                //wait(NULL);
                n = 15;
                printf("Parent Process: %d, Child Process: %d\n", getpid(), pid);
                printf("Parent: n: %d\n", n);
                printf("Terminated Parent\n");
        }

        return 0;
}

int execlp()

  • Create a child process. In the child process, copy input.txt to output.txt
  • From parent print output.txt
#include <stdio.h>
#include <sys/types.h>
#include <unistd.h>
#include <stdlib.h>
#include <sys/wait.h>

int main(int argc, char* argv){
	pid_t pid = fork();
	if(pid == 0){
		FILE* input = fopen("input.txt", "r");
		FILE* output = fopen("output.txt", "w+");

		char c = fgetc(input);
		while(c != EOF){
			fputc(c, output);
			c = fgetc(input);
		}
		fclose(input);
		fclose(output);
		printf("The Child process is now complete\n");
	}else{
		wait(NULL);
		FILE* input = fopen("input.txt", "r");

		char c = fgetc(input);
		while(c != EOF){
			printf("%c", c);
			c = fgetc(input);
		}
	}
}
